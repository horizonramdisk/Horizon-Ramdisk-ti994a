;CCFIXS 8-27-2021 tt
;
;Temporary program to fix CorComp Controller woe due to MDOS not resetting
;      the ROM bank to default during powerup
;
;      See DSRPASS-S: NOTMYR for SBO 11 error
;
       LWPI >F000
       LI   R12,>1100
       SBZ  11

       LI   R0,>27
       LI   R1,FIX1
       CLR  R2
       XOP  @VID,0
       BLWP @0

VID    DATA 6
FIX1   BYTE 10,13
       TEXT 'CorComp FDC reset to bank 0'
       BYTE 10,13,0
